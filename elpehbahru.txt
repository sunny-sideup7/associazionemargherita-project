                        function showSlide(index) {
                            // Hide all main slides
                            const mainSlides = document.querySelectorAll('.main-slide');
                            mainSlides.forEach(slide => {
                                slide.style.display = 'none';
                                // Stop video if it's the video slide being hidden
                                const video = slide.querySelector('video');
                                if (video) {
                                    video.pause();
                                    video.currentTime = 0;
                                }
                            });

                            // Show the selected slide
                            const selectedSlide = document.querySelector(`[data-testid="productDetailImageSlider#${index}"]`);
                            if (selectedSlide) {
                                if (index === 1) { // Video slide
                                    selectedSlide.style.display = 'flex';
                                    const video = selectedSlide.querySelector('video');
                                    if (video) {
                                        video.play();
                                    }
                                } else {
                                    selectedSlide.style.display = 'block';
                                }
                            }

                            // Update thumbnail borders
                            const thumbs = document.querySelectorAll('[data-testid^="productDetailImageSliderSmall#"]');
                            thumbs.forEach((thumb, i) => {
                                thumb.style.border = (i === index) ? '2px solid #007bff' : 'none';
